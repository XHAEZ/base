TRIGGER trg_auditoria_unidad
AFTER INSERT OR UPDATE OR DELETE ON unidad
FOR EACH ROW
DECLARE
    v_accion VARCHAR2(20);
    v_id_unidad NUMBER(10);
    v_cambios VARCHAR2(4000); -- Almacena los cambios realizados
BEGIN
    IF INSERTING THEN
        v_accion := 'INSERCIÓN';
        v_id_unidad := :NEW.id_unidad;
    ELSIF UPDATING THEN
        v_accion := 'ACTUALIZACIÓN';
        v_id_unidad:= :NEW.id_unidad;
    ELSIF DELETING THEN
        v_accion := 'ELIMINACIÓN';
        v_id_unidad := :OLD.id_unidad;
    END IF;

    INSERT INTO auditoria_unidad (id_unidad, accion, usuario)
    VALUES (v_id_unidad, v_accion, USER);
END;
